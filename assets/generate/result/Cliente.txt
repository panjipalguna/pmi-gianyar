    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'id', 'nombre', 'telefono', 'foto');
        $table = "cliente";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->id.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->nombre.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->telefono.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = $this->input->post("txtid"); $txtnombre = $this->input->post("txtnombre"); $txttelefono = $this->input->post("txttelefono"); $txtfoto = $this->input->post("txtfoto"); $txtfoto_hdn = $this->input->post("txtfoto_hdn"); 
        
                $txtfoto =  $txtfoto_hdn;
                if($_FILES['txtfoto']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtfoto']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtfoto'))
                    {
                         $data = $this->upload->data();
                        $txtfoto =  $file_name;
                        if($txtfoto_hdn != '')
                          unlink('uploaded/'.$txtfoto_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'id' => $txtid, 'nombre' => $txtnombre, 'telefono' => $txttelefono, 'foto' => $txtfoto,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'id' => $txtid, 'nombre' => $txtnombre, 'telefono' => $txttelefono, 'foto' => $txtfoto,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
                $rs_cliente = getByQuery("select * from cliente where row_id = ".$txtrowid);
                if($rs_cliente != "") {
                    foreach($rs_cliente as $row) {
                        unlink('uploaded/'.$row->foto); 

                    }
                }
              
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
