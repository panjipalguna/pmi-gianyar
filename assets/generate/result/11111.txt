    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtField Name = $this->input->post("txtField Name"); $txtField Name 2 = $this->input->post("txtField Name 2"); $txtField Name 3 = $this->input->post("txtField Name 3"); $txtField Name 4 = $this->input->post("txtField Name 4"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'Field Name' => $txtField Name, 'Field Name 2' => $txtField Name 2, 'Field Name 3' => $txtField Name 3, 'Field Name 4' => date("Y-m-d H:i:s", strtotime($txtField Name 4)),         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Field Name' => $txtField Name, 'Field Name 2' => $txtField Name 2, 'Field Name 3' => $txtField Name 3, 'Field Name 4' => date("Y-m-d H:i:s", strtotime($txtField Name 4)),              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
