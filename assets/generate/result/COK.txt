            ';

$row[] = '<div style="text-align:center">'.$line->yyyyy.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->yyyy.'</div>';

            $data[] = $row;

            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtyyyyyyyy = $this->input->post("txtyyyyyyyy"); $txtyyyyy = $this->input->post("txtyyyyy"); $txtyyyy = $this->input->post("txtyyyy"); $txtyy = $this->input->post("txtyy"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'yyyyyyyy' => $txtyyyyyyyy, 'yyyyy' => $txtyyyyy, 'yyyy' => $txtyyyy, 'yy' => $txtyy,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'yyyyyyyy' => $txtyyyyyyyy, 'yyyyy' => $txtyyyyy, 'yyyy' => $txtyyyy, 'yy' => $txtyy,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
