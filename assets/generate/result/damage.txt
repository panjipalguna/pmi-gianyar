    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txttransactiondate = $this->input->post("txttransactiondate"); $txttransactiontimehour = $this->input->post("txttransactiontimehour"); $txttransactiontimeminute = $this->input->post("txttransactiontimeminute"); $txtdamagedescription = $this->input->post("txtdamagedescription"); $txtgarageid = $this->input->post("txtgarageid"); $txtfile1 = $this->input->post("txtfile1"); $txtfile1_hdn = $this->input->post("txtfile1_hdn"); $txtfile2 = $this->input->post("txtfile2"); $txtfile2_hdn = $this->input->post("txtfile2_hdn"); $txtfile3 = $this->input->post("txtfile3"); $txtfile3_hdn = $this->input->post("txtfile3_hdn"); $txtfile4 = $this->input->post("txtfile4"); $txtfile4_hdn = $this->input->post("txtfile4_hdn"); 
        
                $txtfile4 =  $txtfile4_hdn;
                if($_FILES['txtfile4']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtfile4']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtfile4'))
                    {
                         $data = $this->upload->data();
                        $txtfile4 =  $file_name;
                        if($txtfile4_hdn != '')
                          unlink('uploaded/'.$txtfile4_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'transaction_date' => date("Y-m-d H:i:s", strtotime($txttransactiondate)), 'transaction_time_hour' => $txttransactiontimehour, 'transaction_time_minute' => $txttransactiontimeminute, 'damage_description' => $txtdamagedescription, 'garage_id' => $txtgarageid, 'file_1' => $txtfile1, 'file_2' => $txtfile2, 'file_3' => $txtfile3, 'file_4' => $txtfile4,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'transaction_date' => date("Y-m-d H:i:s", strtotime($txttransactiondate)), 'transaction_time_hour' => $txttransactiontimehour, 'transaction_time_minute' => $txttransactiontimeminute, 'damage_description' => $txtdamagedescription, 'garage_id' => $txtgarageid, 'file_1' => $txtfile1, 'file_2' => $txtfile2, 'file_3' => $txtfile3, 'file_4' => $txtfile4,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
