    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtTitulo = $this->input->post("txtTitulo"); $txtImagen = $this->input->post("txtImagen"); $txtImagen_hdn = $this->input->post("txtImagen_hdn"); $txtDescripcion = $this->input->post("txtDescripcion"); 
        
                $txtImagen =  $txtImagen_hdn;
                if($_FILES['txtImagen']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtImagen']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtImagen'))
                    {
                         $data = $this->upload->data();
                        $txtImagen =  $file_name;
                        if($txtImagen_hdn != '')
                          unlink('uploaded/'.$txtImagen_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'Titulo' => $txtTitulo, 'Imagen' => $txtImagen, 'Descripcion' => $txtDescripcion,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Titulo' => $txtTitulo, 'Imagen' => $txtImagen, 'Descripcion' => $txtDescripcion,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
