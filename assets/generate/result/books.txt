    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtbookname = $this->input->post("txtbookname"); $txtbookprice = $this->input->post("txtbookprice"); $txtbookimage = $this->input->post("txtbookimage"); $txtbookimage_hdn = $this->input->post("txtbookimage_hdn"); $txtbookowner = $this->input->post("txtbookowner"); 
        
                $txtbookimage =  $txtbookimage_hdn;
                if($_FILES['txtbookimage']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtbookimage']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtbookimage'))
                    {
                         $data = $this->upload->data();
                        $txtbookimage =  $file_name;
                        if($txtbookimage_hdn != '')
                          unlink('uploaded/'.$txtbookimage_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'book_name' => $txtbookname, 'book_price' => $txtbookprice, 'book_image' => $txtbookimage, 'book_owner' => $txtbookowner,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'book_name' => $txtbookname, 'book_price' => $txtbookprice, 'book_image' => $txtbookimage, 'book_owner' => $txtbookowner,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
