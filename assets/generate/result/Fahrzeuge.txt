<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class Fahrzeuge extends CI_Controller {
    function __construct() {
        parent::__construct();
        islogin();
    }

    public $table_db = "wp_Fahrzeuge";
    public $controller = "Fahrzeuge";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            
        );
        $this->layout->display('page/Fahrzeuge_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'id', 'Name', 'Beschreibung', 'Kennzeichen');
        $table = "wp_Fahrzeuge";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->id.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->Name.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->Beschreibung.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->Kennzeichen.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function form() {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = ''; 
$txtName = ''; 
$txtBeschreibung = ''; 
$txtKennzeichen = ''; 


        if($txtstate == "edit") {
            $qry = "select * from wp_Fahrzeuge where row_id = '".$txtrowid."'";
            $rs = getByQuery($qry);
            if($rs != "") {
                $txtid = $rs[0]->id; 
$txtName = $rs[0]->Name; 
$txtBeschreibung = $rs[0]->Beschreibung; 
$txtKennzeichen = $rs[0]->Kennzeichen; 

            }
        }

        

        $table = '
            <div class="row">
                <form method="POST" class="formInput" enctype="multipart/form-data">
                    <input type="hidden" id="txtrowid" name="txtrowid" class="form-control" value="'.$txtrowid.'" style="background:#fff" readonly />
                    <input type="hidden" id="txtstate" name="txtstate" class="form-control" value="'.$txtstate.'" />
                    
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Id:</label>
                        <input type="number" step="1" " id="txtid" name="txtid" class="form-control" value="'.$txtid.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Name:</label>
                        <textarea id="txtName" name="txtName" rows="4" class="form-control">'.$txtName.'</textarea>
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Beshreibung:</label>
                        <textarea id="txtBeschreibung" name="txtBeschreibung" rows="4" class="form-control">'.$txtBeschreibung.'</textarea>
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Kennzeichen:</label>
                        <textarea id="txtKennzeichen" name="txtKennzeichen" rows="4" class="form-control">'.$txtKennzeichen.'</textarea>
                      </div>
                    </div>
                  
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function view() {
        $txtrowid = $this->input->post("txtrowid");
        $txtid = ''; 
$txtName = ''; 
$txtBeschreibung = ''; 
$txtKennzeichen = ''; 

        
        $qry = "select * from wp_Fahrzeuge where row_id = '".$txtrowid."'";
        $rs = getByQuery($qry);
        if($rs != "") {
                $txtid = $rs[0]->id; 
$txtName = $rs[0]->Name; 
$txtBeschreibung = $rs[0]->Beschreibung; 
$txtKennzeichen = $rs[0]->Kennzeichen; 

        }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = $this->input->post("txtid"); $txtName = $this->input->post("txtName"); $txtBeschreibung = $this->input->post("txtBeschreibung"); $txtKennzeichen = $this->input->post("txtKennzeichen"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'id' => $txtid, 'Name' => $txtName, 'Beschreibung' => $txtBeschreibung, 'Kennzeichen' => $txtKennzeichen,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'id' => $txtid, 'Name' => $txtName, 'Beschreibung' => $txtBeschreibung, 'Kennzeichen' => $txtKennzeichen,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
}
