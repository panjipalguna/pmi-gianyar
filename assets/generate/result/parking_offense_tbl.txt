CREATE TABLE parkingoffensetable (
  row_id int(11) NOT NULL AUTO_INCREMENT,
  license_plate VARCHAR(255) NULL, transaction_date DATETIME NULL, transaction_time_hour VARCHAR(255) NULL, transaction_time_minute VARCHAR(255) NULL, parkingviolation_id INT NULL, penalty_id INT NULL, file_1 varchar(20) NULL, file_2 varchar(20) NULL, file_3 varchar(20) NULL, file_4 varchar(20) NULL, 
  PRIMARY KEY (`row_id`), KEY `IDX` (parkingviolation_id,penalty_id)  
) ENGINE=InnoDB DEFAULT CHARSET=latin1;


              create view vw_parkingoffensetablelist as 
              select 
                parkingoffensetable.row_id 
                ,parkingoffensetable.license_plate 
,parkingoffensetable.transaction_date 
,parkingoffensetable.transaction_time_hour 
,parkingoffensetable.transaction_time_minute 
,parkingoffensetable.parkingviolation_id 
,parkingviolationtable.name as parkingviolationtable_name 
,parkingoffensetable.penalty_id 
,penaltytable.name as penaltytable_name 
,parkingoffensetable.file_1 
,parkingoffensetable.file_2 
,parkingoffensetable.file_3 
,parkingoffensetable.file_4 
 
              from parkingoffensetable 
              inner join parkingviolationtable on parkingviolationtable.row_id = parkingoffensetable.parkingviolation_id 
inner join penaltytable on penaltytable.row_id = parkingoffensetable.penalty_id
