    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtblogmetatitle = $this->input->post("txtblogmetatitle"); $txtblogmetadescription = $this->input->post("txtblogmetadescription"); $txtblogmetakeywords = $this->input->post("txtblogmetakeywords"); $txtblogmetarobots = $this->input->post("txtblogmetarobots"); $txtblogmetapagetopic = $this->input->post("txtblogmetapagetopic"); $txtblogmetacopyright = $this->input->post("txtblogmetacopyright"); $txtblogmetaauthor = $this->input->post("txtblogmetaauthor"); $txtblogmetapublisher = $this->input->post("txtblogmetapublisher"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'blog_meta_title' => $txtblogmetatitle, 'blog_meta_description' => $txtblogmetadescription, 'blog_meta_keywords' => $txtblogmetakeywords, 'blog_meta_robots' => $txtblogmetarobots, 'blog_meta_page_topic' => $txtblogmetapagetopic, 'blog_meta_copyright' => $txtblogmetacopyright, 'blog_meta_author' => $txtblogmetaauthor, 'blog_meta_publisher' => $txtblogmetapublisher,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'blog_meta_title' => $txtblogmetatitle, 'blog_meta_description' => $txtblogmetadescription, 'blog_meta_keywords' => $txtblogmetakeywords, 'blog_meta_robots' => $txtblogmetarobots, 'blog_meta_page_topic' => $txtblogmetapagetopic, 'blog_meta_copyright' => $txtblogmetacopyright, 'blog_meta_author' => $txtblogmetaauthor, 'blog_meta_publisher' => $txtblogmetapublisher,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
