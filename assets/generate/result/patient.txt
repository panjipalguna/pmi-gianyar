    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = $this->input->post("txtid"); $txtipp = $this->input->post("txtipp"); $txtNdossier = $this->input->post("txtNdossier"); $txtnomprenom = $this->input->post("txtnomprenom"); $txtdiagnostic = $this->input->post("txtdiagnostic"); $txtCS = $this->input->post("txtCS"); $txtCRO = $this->input->post("txtCRO"); $txtdateentree = $this->input->post("txtdateentree"); $txtdateoperation = $this->input->post("txtdateoperation"); $txtdatesortie = $this->input->post("txtdatesortie"); $txtprofesseur = $this->input->post("txtprofesseur"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'id' => $txtid, 'ipp' => $txtipp, 'N_dossier' => $txtNdossier, 'nom_prenom' => $txtnomprenom, 'diagnostic' => $txtdiagnostic, 'CS' => $txtCS, 'CRO' => $txtCRO, 'date_entree' => date("Y-m-d H:i:s", strtotime($txtdateentree)), 'date_operation' => date("Y-m-d H:i:s", strtotime($txtdateoperation)), 'date_sortie' => date("Y-m-d H:i:s", strtotime($txtdatesortie)), 'professeur' => $txtprofesseur,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'id' => $txtid, 'ipp' => $txtipp, 'N_dossier' => $txtNdossier, 'nom_prenom' => $txtnomprenom, 'diagnostic' => $txtdiagnostic, 'CS' => $txtCS, 'CRO' => $txtCRO, 'date_entree' => date("Y-m-d H:i:s", strtotime($txtdateentree)), 'date_operation' => date("Y-m-d H:i:s", strtotime($txtdateoperation)), 'date_sortie' => date("Y-m-d H:i:s", strtotime($txtdatesortie)), 'professeur' => $txtprofesseur,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
