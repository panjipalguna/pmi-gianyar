CREATE TABLE nome (
  row_id int(11) NOT NULL AUTO_INCREMENT,
  nome INT NULL, 
  PRIMARY KEY (`row_id`), KEY `IDX` (nome)  
) ENGINE=InnoDB DEFAULT CHARSET=latin1;


              create view vw_nomelist as 
              select 
                nome.row_id 
                ,nome.nome 
,nome. as nome_ 
 
              from nome 
              left join nome on nome.nome = nome.nome
