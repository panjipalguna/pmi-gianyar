    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtonlineexamid = $this->input->post("txtonlineexamid"); $txtcode = $this->input->post("txtcode"); $txttitle = $this->input->post("txttitle"); $txtclassid = $this->input->post("txtclassid"); $txtsectionid = $this->input->post("txtsectionid"); $txtsubjectid = $this->input->post("txtsubjectid"); $txtexamdate = $this->input->post("txtexamdate"); $txttimestart = $this->input->post("txttimestart"); $txttimeend = $this->input->post("txttimeend"); $txtduration = $this->input->post("txtduration"); $txtminimumpercentage = $this->input->post("txtminimumpercentage"); $txtinstruction = $this->input->post("txtinstruction"); $txtstatus = $this->input->post("txtstatus"); $txtrunningyear = $this->input->post("txtrunningyear"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'online_exam_id' => $txtonlineexamid, 'code' => $txtcode, 'title' => $txttitle, 'class_id' => $txtclassid, 'section_id' => $txtsectionid, 'subject_id' => $txtsubjectid, 'exam_date' => date("Y-m-d H:i:s", strtotime($txtexamdate)), 'time_start' => $txttimestart, 'time_end' => $txttimeend, 'duration' => $txtduration, 'minimum_percentage' => $txtminimumpercentage, 'instruction' => $txtinstruction, 'status' => $txtstatus, 'running_year' => $txtrunningyear,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'online_exam_id' => $txtonlineexamid, 'code' => $txtcode, 'title' => $txttitle, 'class_id' => $txtclassid, 'section_id' => $txtsectionid, 'subject_id' => $txtsubjectid, 'exam_date' => date("Y-m-d H:i:s", strtotime($txtexamdate)), 'time_start' => $txttimestart, 'time_end' => $txttimeend, 'duration' => $txtduration, 'minimum_percentage' => $txtminimumpercentage, 'instruction' => $txtinstruction, 'status' => $txtstatus, 'running_year' => $txtrunningyear,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
