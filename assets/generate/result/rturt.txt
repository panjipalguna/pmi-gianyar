<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class Rturt extends CI_Controller {
    function __construct() {
        parent::__construct();
    }

    public $table_db = "zurtzurtzu";
    public $controller = "rturt";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            'rs_rtzurt' => getByQuery('select * from rtzurt'),
                'rs_rtzurtzu' => getByQuery('select * from rtzurtzu'),
                
        );
        $this->layout->display('page/rturt_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'rtzur', 'rtzurtzur');
        $table = "vw_zurtzurtzulist";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->rtzurt_zurtzu.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->rtzurtzu_zurtzu.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function form() {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtrtzur = ''; 
$rtzurt_list = "";$txtrtzurtzurtzu = ''; 
$txtrtzurtzur = ''; 
$rtzurtzu_list = "";$txtrtzurtzuzurtzu = ''; 


        if($txtstate == "edit") {
            $qry = "select * from vw_zurtzurtzulist where row_id = '".$txtrowid."'";
            $rs = getByQuery($qry);
            if($rs != "") {
                $txtrtzur = $rs[0]->rtzur; 
$txtrtzurtzurtzu = $rs[0]->rtzurt_zurtzu; 
$txtrtzurtzur = date('Y-m-d', strtotime($rs[0]->rtzurtzur)); 
$txtrtzurtzuzurtzu = $rs[0]->rtzurtzu_zurtzu; 

            }
        }

        
                  $rs_rtzurt = getByQuery("select * from rtzurt");
                  if($rs_rtzurt != "") {
                      foreach($rs_rtzurt as $row_rtzurt) {
                          $selected = ($row_rtzurt->zurtzurt == $txtrtzur) ? "selected" : "";
                          $rtzurt_list .= "<option value='".$row_rtzurt->zurtzurt."' ".$selected." >".$row_rtzurt->zurtzu."</option>";
                      }
                  }
                
                  $rs_rtzurtzu = getByQuery("select * from rtzurtzu");
                  if($rs_rtzurtzu != "") {
                      foreach($rs_rtzurtzu as $row_rtzurtzu) {
                          $selected = ($row_rtzurtzu->rtzurt == $txtrtzurtzur) ? "selected" : "";
                          $rtzurtzu_list .= "<option value='".$row_rtzurtzu->rtzurt."' ".$selected." >".$row_rtzurtzu->zurtzu."</option>";
                      }
                  }
                

        $table = '
            <div class="row">
                <form method="POST" class="formInput" enctype="multipart/form-data">
                    <input type="hidden" id="txtrowid" name="txtrowid" class="form-control" value="'.$txtrowid.'" style="background:#fff" readonly />
                    <input type="hidden" id="txtstate" name="txtstate" class="form-control" value="'.$txtstate.'" />
                    
                  <div class="col-md-12">
                    <div class="form-group">
                      <label>Tzurtzu:</label>
                      <select id="txtrtzur" name="txtrtzur" class="form-control select2"  required>
                        <option value=""></option>
                        '.$rtzurt_list.'
                      </select>
                    </div>
                  </div>
                
                  <div class="col-md-12">
                    <div class="form-group">
                      <label>Tzurtzu:</label>
                      <select id="txtrtzurtzur" name="txtrtzurtzur" class="form-control select2"  required>
                        <option value=""></option>
                        '.$rtzurtzu_list.'
                      </select>
                    </div>
                  </div>
                
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function view() {
        $txtrowid = $this->input->post("txtrowid");
        $txtrtzur = ''; 
$rtzurt_list = "";$txtrtzurtzurtzu = ''; 
$txtrtzurtzur = ''; 
$rtzurtzu_list = "";$txtrtzurtzuzurtzu = ''; 

        
        $qry = "select * from vw_zurtzurtzulist where row_id = '".$txtrowid."'";
        $rs = getByQuery($qry);
        if($rs != "") {
                $txtrtzur = $rs[0]->rtzur; 
$txtrtzurtzurtzu = $rs[0]->rtzurt_zurtzu; 
$txtrtzurtzur = date('Y-m-d', strtotime($rs[0]->rtzurtzur));

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtrtzur = $this->input->post("txtrtzur"); $txtrtzurtzur = $this->input->post("txtrtzurtzur"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'rtzur' => $txtrtzur, 'rtzurtzur' => date("Y-m-d H:i:s", strtotime($txtrtzurtzur)),         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'rtzur' => $txtrtzur, 'rtzurtzur' => date("Y-m-d H:i:s", strtotime($txtrtzurtzur)),              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
}
