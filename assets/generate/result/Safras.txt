class Safras extends CI_Controller {

        parent::__construct();

    public $table_db = "safras";
    public $controller = "Safras";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            'rs_culturas' => getByQuery('select * from culturas'),
                'rs_lavouras' => getByQuery('select * from lavouras'),
                
        );
        $this->layout->display('page/Safras_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'safra_id', 'safra_cultura', 'safra_ano', 'safra_lavoura');
        $table = "vw_safraslist";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->safra_id.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->culturas_cultura_nome.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->safra_ano.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->lavouras_lavoura_nome.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

$txtsafracultura = ''; 
$culturas_list = "";$txtculturasculturanome = ''; 
$txtsafraano = ''; 
$txtsafralavoura = ''; 
$lavouras_list = "";$txtlavouraslavouranome = ''; 


        if($txtstate == "edit") {
            $qry = "select * from vw_safraslist where row_id = '".$txtrowid."'";

                $txtsafraid = $rs[0]->safra_id;

        }

        
                  $rs_culturas = getByQuery("select * from culturas");

                        <label>: <span style="color: #ff0000">*</span></label>
                        <input type="number" step="1" " id="txtsafraid" name="txtsafraid" class="form-control" value="'.$txtsafraid.'" />
                      </div>
                    </div>
                  
                  <div class="col-md-12">
                    <div class="form-group">
                      <label>Cultura:</label>
                      <select id="txtsafracultura" name="txtsafracultura" class="form-control select2"  required>
                        <option value=""></option>

                    </div>
                  </div>
                
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Ano:</label>
                        <input type="text" id="txtsafraano" name="txtsafraano" class="form-control" value="'.$txtsafraano.'" />

                    <div class="form-group">

                      <select id="txtsafralavoura" name="txtsafralavoura" class="form-control select2"  required>
                        <option value=""></option>
                        '.$lavouras_list.'
                      </select>
                    </div>
                  </div>
                
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtsafraid = $this->input->post("txtsafraid"); $txtsafracultura = $this->input->post("txtsafracultura"); $txtsafraano = $this->input->post("txtsafraano"); $txtsafralavoura = $this->input->post("txtsafralavoura"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'safra_id' => $txtsafraid, 'safra_cultura' => $txtsafracultura, 'safra_ano' => $txtsafraano, 'safra_lavoura' => $txtsafralavoura,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'safra_id' => $txtsafraid, 'safra_cultura' => $txtsafracultura, 'safra_ano' => $txtsafraano, 'safra_lavoura' => $txtsafralavoura,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
