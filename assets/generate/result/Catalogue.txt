    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtIDCatalogue = $this->input->post("txtIDCatalogue"); $txtPays = $this->input->post("txtPays"); $txtIDYvert = $this->input->post("txtIDYvert"); $txtDescription = $this->input->post("txtDescription"); $txtValeurFaciale = $this->input->post("txtValeurFaciale"); $txtAnnee = $this->input->post("txtAnnee"); $txtNonEmis = $this->input->post("txtNonEmis"); $txtPhoto = $this->input->post("txtPhoto"); $txtPhoto_hdn = $this->input->post("txtPhoto_hdn"); 
        
                $txtPhoto =  $txtPhoto_hdn;
                if($_FILES['txtPhoto']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtPhoto']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtPhoto'))
                    {
                         $data = $this->upload->data();
                        $txtPhoto =  $file_name;
                        if($txtPhoto_hdn != '')
                          unlink('uploaded/'.$txtPhoto_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'IDCatalogue' => $txtIDCatalogue, 'Pays' => $txtPays, 'IDYvert' => $txtIDYvert, 'Description' => $txtDescription, 'ValeurFaciale' => $txtValeurFaciale, 'Annee' => $txtAnnee, 'NonEmis' => $txtNonEmis, 'Photo' => $txtPhoto,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'IDCatalogue' => $txtIDCatalogue, 'Pays' => $txtPays, 'IDYvert' => $txtIDYvert, 'Description' => $txtDescription, 'ValeurFaciale' => $txtValeurFaciale, 'Annee' => $txtAnnee, 'NonEmis' => $txtNonEmis, 'Photo' => $txtPhoto,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
                $rs_catalogue = getByQuery("select * from catalogue where row_id = ".$txtrowid);
                if($rs_catalogue != "") {
                    foreach($rs_catalogue as $row) {
                        unlink('uploaded/'.$row->Photo); 

                    }
                }
              
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
