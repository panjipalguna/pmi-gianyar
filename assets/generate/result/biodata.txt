    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = $this->input->post("txtid"); $txtname = $this->input->post("txtname"); $txtimage1 = $this->input->post("txtimage1"); $txtimage1_hdn = $this->input->post("txtimage1_hdn"); $txtimage2 = $this->input->post("txtimage2"); $txtimage2_hdn = $this->input->post("txtimage2_hdn"); $txtmobile = $this->input->post("txtmobile"); $txtemail = $this->input->post("txtemail"); $txtaddress = $this->input->post("txtaddress"); $txtdob = $this->input->post("txtdob"); 
        
                $txtimage2 =  $txtimage2_hdn;
                if($_FILES['txtimage2']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtimage2']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtimage2'))
                    {
                         $data = $this->upload->data();
                        $txtimage2 =  $file_name;
                        if($txtimage2_hdn != '')
                          unlink('uploaded/'.$txtimage2_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'id' => $txtid, 'name' => $txtname, 'image1' => $txtimage1, 'image2' => $txtimage2, 'mobile' => $txtmobile, 'email' => $txtemail, 'address' => $txtaddress, 'dob' => date("Y-m-d H:i:s", strtotime($txtdob)),         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'id' => $txtid, 'name' => $txtname, 'image1' => $txtimage1, 'image2' => $txtimage2, 'mobile' => $txtmobile, 'email' => $txtemail, 'address' => $txtaddress, 'dob' => date("Y-m-d H:i:s", strtotime($txtdob)),              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
