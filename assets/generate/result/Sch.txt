    public function gridview() {

        {
            $no++;

                  </ul>
                </div>
            ';

$row[] = '<div style="text-align:center">'.$line->Username.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->Password.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),

    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtName = $this->input->post("txtName"); $txtUsername = $this->input->post("txtUsername"); $txtPassword = $this->input->post("txtPassword"); $txtpassport = $this->input->post("txtpassport"); $txtpassport_hdn = $this->input->post("txtpassport_hdn"); 
        
                $txtpassport =  $txtpassport_hdn;
                if($_FILES['txtpassport']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtpassport']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtpassport'))
                    {
                         $data = $this->upload->data();
                        $txtpassport =  $file_name;
                        if($txtpassport_hdn != '')
                          unlink('uploaded/'.$txtpassport_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'Name' => $txtName, 'Username' => $txtUsername, 'Password' => $txtPassword, 'passport' => $txtpassport,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Name' => $txtName, 'Username' => $txtUsername, 'Password' => $txtPassword, 'passport' => $txtpassport,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
                $rs_Student = getByQuery("select * from Student where row_id = ".$txtrowid);
                if($rs_Student != "") {
                    foreach($rs_Student as $row) {
                        unlink('uploaded/'.$row->passport); 

                    }
                }
              
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
