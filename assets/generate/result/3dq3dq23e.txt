    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtqwdqwed = $this->input->post("txtqwdqwed"); $txtqwdqwed_hdn = $this->input->post("txtqwdqwed_hdn"); 
        
                $txtqwdqwed =  $txtqwdqwed_hdn;
                if($_FILES['txtqwdqwed']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtqwdqwed']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtqwdqwed'))
                    {
                         $data = $this->upload->data();
                        $txtqwdqwed =  $file_name;
                        if($txtqwdqwed_hdn != '')
                          unlink('uploaded/'.$txtqwdqwed_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'qwdqwed' => $txtqwdqwed,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'qwdqwed' => $txtqwdqwed,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
