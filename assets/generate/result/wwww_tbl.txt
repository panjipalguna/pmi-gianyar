CREATE TABLE wwww (
  row_id int(11) NOT NULL AUTO_INCREMENT,
  awdw INT NULL, 
  PRIMARY KEY (`row_id`), KEY `IDX` (awdw)  
) ENGINE=InnoDB DEFAULT CHARSET=latin1;


              create view vw_wwwwlist as 
              select 
                wwww.row_id 
                ,wwww.awdw 
,a.a as a_a 
 
              from wwww 
              left join a on a.a = wwww.awdw
