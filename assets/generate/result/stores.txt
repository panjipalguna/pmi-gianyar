    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtTitle = $this->input->post("txtTitle"); $txtShoplink = $this->input->post("txtShoplink"); $txtCouponlink = $this->input->post("txtCouponlink"); $txtCircularlink = $this->input->post("txtCircularlink"); $txtStorecard = $this->input->post("txtStorecard"); $txtLogo = $this->input->post("txtLogo"); $txtLogo_hdn = $this->input->post("txtLogo_hdn"); $txtidsub = $this->input->post("txtidsub"); 
        
                $txtLogo =  $txtLogo_hdn;
                if($_FILES['txtLogo']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtLogo']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtLogo'))
                    {
                         $data = $this->upload->data();
                        $txtLogo =  $file_name;
                        if($txtLogo_hdn != '')
                          unlink('uploaded/'.$txtLogo_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'Title' => $txtTitle, 'Shop_link' => $txtShoplink, 'Coupon_link' => $txtCouponlink, 'Circular_link' => $txtCircularlink, 'Store_card' => $txtStorecard, 'Logo' => $txtLogo, 'id_sub' => $txtidsub,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Title' => $txtTitle, 'Shop_link' => $txtShoplink, 'Coupon_link' => $txtCouponlink, 'Circular_link' => $txtCircularlink, 'Store_card' => $txtStorecard, 'Logo' => $txtLogo, 'id_sub' => $txtidsub,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
