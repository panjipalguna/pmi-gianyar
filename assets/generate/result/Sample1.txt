    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtt1id = $this->input->post("txtt1id"); $txtt1fk1 = $this->input->post("txtt1fk1"); $txtt1fk2 = $this->input->post("txtt1fk2"); $txtt1field = $this->input->post("txtt1field"); $txtt1img = $this->input->post("txtt1img"); $txtt1img_hdn = $this->input->post("txtt1img_hdn"); $txtt1date = $this->input->post("txtt1date"); $txtt1fieeld2 = $this->input->post("txtt1fieeld2"); 
        
                $txtt1img =  $txtt1img_hdn;
                if($_FILES['txtt1img']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtt1img']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtt1img'))
                    {
                         $data = $this->upload->data();
                        $txtt1img =  $file_name;
                        if($txtt1img_hdn != '')
                          unlink('uploaded/'.$txtt1img_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                't1_id' => $txtt1id, 't1_fk1' => $txtt1fk1, 't1_fk2' => $txtt1fk2, 't1_field' => $txtt1field, 't1_img' => $txtt1img, 't1_date' => date("Y-m-d H:i:s", strtotime($txtt1date)), 't1_fieeld2' => $txtt1fieeld2,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                't1_id' => $txtt1id, 't1_fk1' => $txtt1fk1, 't1_fk2' => $txtt1fk2, 't1_field' => $txtt1field, 't1_img' => $txtt1img, 't1_date' => date("Y-m-d H:i:s", strtotime($txtt1date)), 't1_fieeld2' => $txtt1fieeld2,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
