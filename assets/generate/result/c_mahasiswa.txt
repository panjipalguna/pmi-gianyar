<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class C_mahasiswa extends CI_Controller {
    function __construct() {
        parent::__construct();
        islogin();
    }

    public $table_db = "mahasiswa";
    public $controller = "c_mahasiswa";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            
        );
        $this->layout->display('page/c_mahasiswa_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'npm', 'nama', 'jenis_kelamin', 'almaat');
        $table = "mahasiswa";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->npm.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->nama.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->jenis_kelamin.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->almaat.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function form() {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtnpm = ''; 
$txtnama = ''; 
$txtjeniskelamin = ''; 
$txtalmaat = ''; 


        if($txtstate == "edit") {
            $qry = "select * from mahasiswa where row_id = '".$txtrowid."'";
            $rs = getByQuery($qry);
            if($rs != "") {
                $txtnpm = $rs[0]->npm; 
$txtnama = $rs[0]->nama; 
$txtjeniskelamin = $rs[0]->jenis_kelamin; 
$txtalmaat = $rs[0]->almaat; 

            }
        }

        

        $table = '
            <div class="row">
                <form method="POST" class="formInput" enctype="multipart/form-data">
                    <input type="hidden" id="txtrowid" name="txtrowid" class="form-control" value="'.$txtrowid.'" style="background:#fff" readonly />
                    <input type="hidden" id="txtstate" name="txtstate" class="form-control" value="'.$txtstate.'" />
                    
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Npm Mahasiswa:</label>
                        <input type="text" id="txtnpm" name="txtnpm" class="form-control" value="'.$txtnpm.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Nama Mahasiswa:</label>
                        <input type="text" id="txtnama" name="txtnama" class="form-control" value="'.$txtnama.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Jenis Kelamin:</label>
                        <input type="text" id="txtjeniskelamin" name="txtjeniskelamin" class="form-control" value="'.$txtjeniskelamin.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Alamat:</label>
                        <textarea id="txtalmaat" name="txtalmaat" rows="4" class="form-control">'.$txtalmaat.'</textarea>
                      </div>
                    </div>
                  
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function view() {
        $txtrowid = $this->input->post("txtrowid");
        $txtnpm = ''; 
$txtnama = ''; 
$txtjeniskelamin = ''; 
$txtalmaat = ''; 

        
        $qry = "select * from mahasiswa where row_id = '".$txtrowid."'";
        $rs = getByQuery($qry);
        if($rs != "") {
                $txtnpm = $rs[0]->npm; 
$txtnama = $rs[0]->nama; 
$txtjeniskelamin = $rs[0]->jenis_kelamin; 
$txtalmaat = $rs[0]->almaat; 

        }        

        $table = '
            <table id="tablelist" class="table table-striped">
              <tbody>
                
                    <tr>
                        <td class="view-title" style="width: 30%">Npm Mahasiswa:</td>
                        <td class="view-txt">'.$txtnpm.'</td>

                    <tr>
                        <td class="view-title" style="width: 30%">Nama Mahasiswa:</td>
                        <td class="view-txt">'.$txtnama.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">Jenis Kelamin:</td>
                        <td class="view-txt">'.$txtjeniskelamin.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">Alamat:</td>
                        <td class="view-txt">'.$txtalmaat.'</td>
                    </tr>
                  
              </tbody>
            </table>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtnpm = $this->input->post("txtnpm"); $txtnama = $this->input->post("txtnama"); $txtjeniskelamin = $this->input->post("txtjeniskelamin"); $txtalmaat = $this->input->post("txtalmaat"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'npm' => $txtnpm, 'nama' => $txtnama, 'jenis_kelamin' => $txtjeniskelamin, 'almaat' => $txtalmaat,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'npm' => $txtnpm, 'nama' => $txtnama, 'jenis_kelamin' => $txtjeniskelamin, 'almaat' => $txtalmaat,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
}
