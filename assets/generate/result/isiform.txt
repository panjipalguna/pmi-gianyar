    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtfile1 = $this->input->post("txtfile1"); $txtfile1_hdn = $this->input->post("txtfile1_hdn"); $txtkodebr = $this->input->post("txtkodebr"); 
        
                $txtfile1 =  $txtfile1_hdn;
                if($_FILES['txtfile1']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtfile1']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtfile1'))
                    {
                         $data = $this->upload->data();
                        $txtfile1 =  $file_name;
                        if($txtfile1_hdn != '')
                          unlink('uploaded/'.$txtfile1_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'file1' => $txtfile1, 'kode_br' => $txtkodebr,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'file1' => $txtfile1, 'kode_br' => $txtkodebr,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
