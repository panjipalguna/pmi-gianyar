    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtnomprenom = $this->input->post("txtnomprenom"); $txtipp = $this->input->post("txtipp"); $txtndossier = $this->input->post("txtndossier"); $txtdiagnosticid = $this->input->post("txtdiagnosticid"); $txtprofesseurid = $this->input->post("txtprofesseurid"); $txtdateentree = $this->input->post("txtdateentree"); $txtlit = $this->input->post("txtlit"); $txtdateoperation = $this->input->post("txtdateoperation"); $txtdatesortie = $this->input->post("txtdatesortie"); $txtCS = $this->input->post("txtCS"); $txtCS_hdn = $this->input->post("txtCS_hdn"); $txtCRO = $this->input->post("txtCRO"); $txtCRO_hdn = $this->input->post("txtCRO_hdn"); $txtobs = $this->input->post("txtobs"); 
        
                $txtCRO =  $txtCRO_hdn;
                if($_FILES['txtCRO']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtCRO']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtCRO'))
                    {
                         $data = $this->upload->data();
                        $txtCRO =  $file_name;
                        if($txtCRO_hdn != '')
                          unlink('uploaded/'.$txtCRO_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'nom_prenom' => $txtnomprenom, 'ipp' => $txtipp, 'n_dossier' => $txtndossier, 'diagnostic_id' => $txtdiagnosticid, 'professeur_id' => $txtprofesseurid, 'date_entree' => date("Y-m-d H:i:s", strtotime($txtdateentree)), 'lit' => $txtlit, 'date_operation' => date("Y-m-d H:i:s", strtotime($txtdateoperation)), 'date_sortie' => date("Y-m-d H:i:s", strtotime($txtdatesortie)), 'CS' => $txtCS, 'CRO' => $txtCRO, 'obs' => $txtobs,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'nom_prenom' => $txtnomprenom, 'ipp' => $txtipp, 'n_dossier' => $txtndossier, 'diagnostic_id' => $txtdiagnosticid, 'professeur_id' => $txtprofesseurid, 'date_entree' => date("Y-m-d H:i:s", strtotime($txtdateentree)), 'lit' => $txtlit, 'date_operation' => date("Y-m-d H:i:s", strtotime($txtdateoperation)), 'date_sortie' => date("Y-m-d H:i:s", strtotime($txtdatesortie)), 'CS' => $txtCS, 'CRO' => $txtCRO, 'obs' => $txtobs,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
