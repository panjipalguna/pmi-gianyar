    public function gridview() {
        $where = $this->input->post("where");

        $column = array('row_id', 'name', 'salary', 'date', 'image', 'family_number');
        $table = "customer";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">

                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->name.'</div>'; 
$row[] = '<div style="text-align:right">'.number_format($line->salary).'</div>'; 
$row[] = '<div style="text-align:center">'.date("m/d/Y", strtotime($line->date)).'</div>'; 
$row[] = '<div style="text-align:center">'.$line->family_number.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtname = $this->input->post("txtname"); $txtsalary = $this->input->post("txtsalary"); $txtdate = $this->input->post("txtdate"); $txtimage = $this->input->post("txtimage"); $txtimage_hdn = $this->input->post("txtimage_hdn"); $txtfamilynumber = $this->input->post("txtfamilynumber"); 
        
                $txtimage =  $txtimage_hdn;
                if($_FILES['txtimage']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtimage']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtimage'))
                    {
                         $data = $this->upload->data();
                        $txtimage =  $file_name;
                        if($txtimage_hdn != '')
                          unlink('uploaded/'.$txtimage_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'name' => $txtname, 'salary' => $txtsalary, 'date' => date("Y-m-d H:i:s", strtotime($txtdate)), 'image' => $txtimage, 'family_number' => $txtfamilynumber,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'name' => $txtname, 'salary' => $txtsalary, 'date' => date("Y-m-d H:i:s", strtotime($txtdate)), 'image' => $txtimage, 'family_number' => $txtfamilynumber,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
