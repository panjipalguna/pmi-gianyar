CREATE TABLE online_exam (
  row_id int(11) NOT NULL AUTO_INCREMENT,
  online_exam_id INT NULL, code VARCHAR(255) NULL, title TEXT NULL, class_id INT NULL, section_id INT NULL, subject_id INT NULL, exam_date DATETIME NULL, time_start VARCHAR(255) NULL, time_end VARCHAR(255) NULL, duration TEXT NULL, minimum_percentage TEXT NULL, instruction TEXT NULL, status VARCHAR(255) NULL, running_year VARCHAR(255) NULL, 
  PRIMARY KEY (`row_id`)  
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
