    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtfile1 = $this->input->post("txtfile1"); $txtfile1_hdn = $this->input->post("txtfile1_hdn"); $txtdept = $this->input->post("txtdept"); $txtkoderevisi = $this->input->post("txtkoderevisi"); $txtfile2 = $this->input->post("txtfile2"); $txtfile2_hdn = $this->input->post("txtfile2_hdn"); 
        
                $txtfile2 =  $txtfile2_hdn;
                if($_FILES['txtfile2']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtfile2']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtfile2'))
                    {
                         $data = $this->upload->data();
                        $txtfile2 =  $file_name;
                        if($txtfile2_hdn != '')
                          unlink('uploaded/'.$txtfile2_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'file1' => $txtfile1, 'dept' => $txtdept, 'koderevisi' => $txtkoderevisi, 'file2' => $txtfile2,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'file1' => $txtfile1, 'dept' => $txtdept, 'koderevisi' => $txtkoderevisi, 'file2' => $txtfile2,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
