    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtid = $this->input->post("txtid"); $txtImagem = $this->input->post("txtImagem"); $txtImagem_hdn = $this->input->post("txtImagem_hdn"); $txtnome = $this->input->post("txtnome"); 
        
                $txtImagem =  $txtImagem_hdn;
                if($_FILES['txtImagem']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtImagem']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtImagem'))
                    {
                         $data = $this->upload->data();
                        $txtImagem =  $file_name;
                        if($txtImagem_hdn != '')
                          unlink('uploaded/'.$txtImagem_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'id' => $txtid, 'Imagem' => $txtImagem, 'nome' => $txtnome,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'id' => $txtid, 'Imagem' => $txtImagem, 'nome' => $txtnome,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
