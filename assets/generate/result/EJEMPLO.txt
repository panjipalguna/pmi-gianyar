    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'NOMRE', 'CELULAR', 'CEDULA', 'FECHA DE NACIMIENTO', 'PAGO');
        $table = "LUCES";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:right">'.number_format($line->NOMRE).'</div>'; 
$row[] = '<div style="text-align:right">'.number_format($line->CELULAR).'</div>'; 
$row[] = '<div style="text-align:left">'.$line->CEDULA.'</div>'; 
$row[] = '<div style="text-align:center">'.date("m/d/Y", strtotime($line->FECHA DE NACIMIENTO)).'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtNOMRE = $this->input->post("txtNOMRE"); $txtCELULAR = $this->input->post("txtCELULAR"); $txtCEDULA = $this->input->post("txtCEDULA"); $txtFECHA DE NACIMIENTO = $this->input->post("txtFECHA DE NACIMIENTO"); $txtPAGO = $this->input->post("txtPAGO"); $txtPAGO_hdn = $this->input->post("txtPAGO_hdn"); 
        
                $txtPAGO =  $txtPAGO_hdn;
                if($_FILES['txtPAGO']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtPAGO']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtPAGO'))
                    {
                         $data = $this->upload->data();
                        $txtPAGO =  $file_name;
                        if($txtPAGO_hdn != '')
                          unlink('uploaded/'.$txtPAGO_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'NOMRE' => $txtNOMRE, 'CELULAR' => $txtCELULAR, 'CEDULA' => $txtCEDULA, 'FECHA DE NACIMIENTO' => date("Y-m-d H:i:s", strtotime($txtFECHA DE NACIMIENTO)), 'PAGO' => $txtPAGO,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'NOMRE' => $txtNOMRE, 'CELULAR' => $txtCELULAR, 'CEDULA' => $txtCEDULA, 'FECHA DE NACIMIENTO' => date("Y-m-d H:i:s", strtotime($txtFECHA DE NACIMIENTO)), 'PAGO' => $txtPAGO,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
                $rs_LUCES = getByQuery("select * from LUCES where row_id = ".$txtrowid);
                if($rs_LUCES != "") {
                    foreach($rs_LUCES as $row) {
                        unlink('uploaded/'.$row->PAGO); 

                    }
                }
              
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
