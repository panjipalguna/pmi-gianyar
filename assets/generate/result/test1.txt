    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txt1 = $this->input->post("txt1"); $txt2 = $this->input->post("txt2"); $txt3 = $this->input->post("txt3"); $txt4 = $this->input->post("txt4"); $txt5 = $this->input->post("txt5"); $txt5_hdn = $this->input->post("txt5_hdn"); 
        
                $txt5 =  $txt5_hdn;
                if($_FILES['txt5']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txt5']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txt5'))
                    {
                         $data = $this->upload->data();
                        $txt5 =  $file_name;
                        if($txt5_hdn != '')
                          unlink('uploaded/'.$txt5_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                '1' => $txt1, '2' => $txt2, '3' => date("Y-m-d H:i:s", strtotime($txt3)), '4' => $txt4, '5' => $txt5,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                '1' => $txt1, '2' => $txt2, '3' => date("Y-m-d H:i:s", strtotime($txt3)), '4' => $txt4, '5' => $txt5,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
                $rs_test1 = getByQuery("select * from test1 where row_id = ".$txtrowid);

                }
              
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
