<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class Product extends CI_Controller {
    function __construct() {
        parent::__construct();
        islogin();
    }

    public $table_db = "Products";
    public $controller = "Product";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            
        );
        $this->layout->display('page/Product_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'product_id', 'product_name', 'qty');
        $table = "Products";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->product_id.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->product_name.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->qty.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function form() {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtproductid = ''; 
$txtproductname = ''; 
$txtqty = ''; 


        if($txtstate == "edit") {
            $qry = "select * from Products where row_id = '".$txtrowid."'";
            $rs = getByQuery($qry);
            if($rs != "") {
                $txtproductid = $rs[0]->product_id; 
$txtproductname = $rs[0]->product_name; 
$txtqty = $rs[0]->qty; 

            }
        }

        

        $table = '
            <div class="row">
                <form method="POST" class="formInput" enctype="multipart/form-data">
                    <input type="hidden" id="txtrowid" name="txtrowid" class="form-control" value="'.$txtrowid.'" style="background:#fff" readonly />
                    <input type="hidden" id="txtstate" name="txtstate" class="form-control" value="'.$txtstate.'" />
                    
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Product Id: <span style="color: #ff0000">*</span></label>
                        <input type="number" step="1" " id="txtproductid" name="txtproductid" class="form-control" value="'.$txtproductid.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Product Name: <span style="color: #ff0000">*</span></label>
                        <input type="text" id="txtproductname" name="txtproductname" class="form-control" value="'.$txtproductname.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>Qty:</label>
                        <input type="number" step="1" " id="txtqty" name="txtqty" class="form-control" value="'.$txtqty.'" />
                      </div>
                    </div>
                  
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function view() {
        $txtrowid = $this->input->post("txtrowid");
        $txtproductid = ''; 
$txtproductname = ''; 
$txtqty = ''; 

        
        $qry = "select * from Products where row_id = '".$txtrowid."'";
        $rs = getByQuery($qry);
        if($rs != "") {
                $txtproductid = $rs[0]->product_id; 
$txtproductname = $rs[0]->product_name; 
$txtqty = $rs[0]->qty; 

        }        

        $table = '
            <table id="tablelist" class="table table-striped">
              <tbody>
                
                    <tr>
                        <td class="view-title" style="width: 30%">Product Id:</td>
                        <td class="view-txt">'.$txtproductid.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">Product Name:</td>
                        <td class="view-txt">'.$txtproductname.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">Qty:</td>
                        <td class="view-txt">'.$txtqty.'</td>
                    </tr>
                  
              </tbody>
            </table>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtproductid = $this->input->post("txtproductid"); $txtproductname = $this->input->post("txtproductname"); $txtqty = $this->input->post("txtqty"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'product_id' => $txtproductid, 'product_name' => $txtproductname, 'qty' => $txtqty,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'product_id' => $txtproductid, 'product_name' => $txtproductname, 'qty' => $txtqty,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
}
