    public function save()
    {
        $txtstate = $this->input->post("txtstate");

        $txtArtist Name  = $this->input->post("txtArtist Name "); $txtArtist Bio = $this->input->post("txtArtist Bio"); $txtArtist Label  = $this->input->post("txtArtist Label "); $txtArtist Inage = $this->input->post("txtArtist Inage"); $txtArtist Inage_hdn = $this->input->post("txtArtist Inage_hdn"); $txtArtist Location  = $this->input->post("txtArtist Location "); 
        
                $txtArtist Inage =  $txtArtist Inage_hdn;
                if($_FILES['txtArtist Inage']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtArtist Inage']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtArtist Inage'))
                    {
                         $data = $this->upload->data();
                        $txtArtist Inage =  $file_name;
                        if($txtArtist Inage_hdn != '')
                          unlink('uploaded/'.$txtArtist Inage_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'Artist Name ' => $txtArtist Name , 'Artist Bio' => $txtArtist Bio, 'Artist Label ' => $txtArtist Label , 'Artist Inage' => $txtArtist Inage, 'Artist Location ' => $txtArtist Location ,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Artist Name ' => $txtArtist Name , 'Artist Bio' => $txtArtist Bio, 'Artist Label ' => $txtArtist Label , 'Artist Inage' => $txtArtist Inage, 'Artist Location ' => $txtArtist Location ,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
