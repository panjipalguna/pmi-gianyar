<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class Ticket extends CI_Controller {
    function __construct() {
        parent::__construct();
        islogin();
    }

    public $table_db = "ticket";
    public $controller = "ticket";
    
    public function index() {
        $lbl_controller = str_replace("_", " ", $this->controller);
        $page_url = site_url($this->controller);
		$data = array(
            'controller' => $this->controller,
            'lbl_controller' => $lbl_controller,
            'page_url' => $page_url,
            
        );
        $this->layout->display('page/ticket_vw', $data);
    }

    public function gridview() {
        $where = $this->input->post("where");        
        $database = "default";
        $column = array('row_id', 'ticket_id', 'ticket_title', 'ticket_content', 'ticket_status');
        $table = "ticket";
        $list = $this->datatables_mdl->get_datatables($database, $table, $column, $where);
        $data = array();
        $no = $_POST['start'];
        foreach ($list as $line) 
        {
            $no++;          
            $row = array();
            $link = '
                <div class="btn-group">
                  <a href="javascript:void(0)" data-toggle="dropdown" style="color:#111"><i class="fa fa-folder-open"></i></a>
                  <ul class="dropdown-menu" role="menu">
                    <li><a href="javascript:void(0)" style="color:#111" data-toggle="modal" data-target="#modalForm" onclick="generateModalView(\''.$line->row_id.'\')">View</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="generateModalForm(\'edit\', \''.$line->row_id.'\')">Update</a></li>
                    <li><a href="javascript:void(0)" style="color:#111" onclick="doDelete(\''.$line->row_id.'\')">Delete</a></li>
                  </ul>
                </div>
            ';
            $row[] = '<div style="text-align:center">'.$link.'</div>';
            $row[] = '<div style="text-align:center">'.$line->ticket_id.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->ticket_title.'</div>'; 
$row[] = '<div style="text-align:left">'.$line->ticket_content.'</div>'; 
$row[] = '<div style="text-align:center">'.$line->ticket_status.'</div>'; 

            $data[] = $row;
        }
 
        $output = array(
            "draw" => $_POST['draw'],
            "recordsTotal" => $this->datatables_mdl->count_all($database, $table, $column),
            "recordsFiltered" => $this->datatables_mdl->count_filtered($database, $table, $column, $where),
            "data" => $data,
        );
        
        echo json_encode($output);
    }

    public function form() {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtticketid = ''; 
$txttickettitle = ''; 
$txtticketcontent = ''; 
$txtticketstatus = ''; 


        if($txtstate == "edit") {
            $qry = "select * from ticket where row_id = '".$txtrowid."'";
            $rs = getByQuery($qry);
            if($rs != "") {
                $txtticketid = $rs[0]->ticket_id; 
$txttickettitle = $rs[0]->ticket_title; 
$txtticketcontent = $rs[0]->ticket_content; 
$txtticketstatus = $rs[0]->ticket_status; 

            }
        }

        

        $table = '
            <div class="row">

                    <input type="hidden" id="txtrowid" name="txtrowid" class="form-control" value="'.$txtrowid.'" style="background:#fff" readonly />
                    <input type="hidden" id="txtstate" name="txtstate" class="form-control" value="'.$txtstate.'" />
                    
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>: <span style="color: #ff0000">*</span></label>
                        <input type="number" step="1" " id="txtticketid" name="txtticketid" class="form-control" value="'.$txtticketid.'" />
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>: <span style="color: #ff0000">*</span></label>
                        <textarea id="txttickettitle" name="txttickettitle" rows="4" class="form-control">'.$txttickettitle.'</textarea>
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>: <span style="color: #ff0000">*</span></label>
                        <textarea id="txtticketcontent" name="txtticketcontent" rows="4" class="form-control">'.$txtticketcontent.'</textarea>
                      </div>
                    </div>
                  
                    <div class="col-md-12">
                      <div class="form-group">
                        <label>:</label>
                        <input type="number" step="1" " id="txtticketstatus" name="txtticketstatus" class="form-control" value="'.$txtticketstatus.'" />
                      </div>
                    </div>
                  
                </form>
            </div>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function view() {
        $txtrowid = $this->input->post("txtrowid");
        $txtticketid = ''; 
$txttickettitle = ''; 
$txtticketcontent = ''; 
$txtticketstatus = ''; 

        
        $qry = "select * from ticket where row_id = '".$txtrowid."'";
        $rs = getByQuery($qry);
        if($rs != "") {
                $txtticketid = $rs[0]->ticket_id; 
$txttickettitle = $rs[0]->ticket_title; 
$txtticketcontent = $rs[0]->ticket_content; 
$txtticketstatus = $rs[0]->ticket_status; 

        }        

        $table = '
            <table id="tablelist" class="table table-striped">
              <tbody>
                
                    <tr>
                        <td class="view-title" style="width: 30%">:</td>
                        <td class="view-txt">'.$txtticketid.'</td>

                    <tr>
                        <td class="view-title" style="width: 30%">:</td>
                        <td class="view-txt">'.$txttickettitle.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">:</td>
                        <td class="view-txt">'.$txtticketcontent.'</td>
                    </tr>
                  
                    <tr>
                        <td class="view-title" style="width: 30%">:</td>
                        <td class="view-txt">'.$txtticketstatus.'</td>
                    </tr>
                  
              </tbody>
            </table>
        ';

        $data['table'] = $table;
        echo json_encode($data);
    }

    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtticketid = $this->input->post("txtticketid"); $txttickettitle = $this->input->post("txttickettitle"); $txtticketcontent = $this->input->post("txtticketcontent"); $txtticketstatus = $this->input->post("txtticketstatus"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'ticket_id' => $txtticketid, 'ticket_title' => $txttickettitle, 'ticket_content' => $txtticketcontent, 'ticket_status' => $txtticketstatus,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'ticket_id' => $txtticketid, 'ticket_title' => $txttickettitle, 'ticket_content' => $txtticketcontent, 'ticket_status' => $txtticketstatus,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");
        
        deleteData($this->table_db, "row_id = '".$txtrowid."'");
        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
    }
}
