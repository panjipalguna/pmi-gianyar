    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txt dsfds = $this->input->post("txt dsfds"); $txtfdsf = $this->input->post("txtfdsf"); $txtbbbb = $this->input->post("txtbbbb"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                ' dsfds' => $txt dsfds, 'fdsf' => $txtfdsf, 'bbbb' => date("Y-m-d H:i:s", strtotime($txtbbbb)),         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                ' dsfds' => $txt dsfds, 'fdsf' => $txtfdsf, 'bbbb' => date("Y-m-d H:i:s", strtotime($txtbbbb)),              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
