    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtreleasename = $this->input->post("txtreleasename"); $txtreleasedate = $this->input->post("txtreleasedate"); $txtreleaseartwork = $this->input->post("txtreleaseartwork"); $txtreleaseartwork_hdn = $this->input->post("txtreleaseartwork_hdn"); $txtreleaselabel = $this->input->post("txtreleaselabel"); $txtreleaseupc = $this->input->post("txtreleaseupc"); $txtreleasetaken = $this->input->post("txtreleasetaken"); $txtreleaseid = $this->input->post("txtreleaseid"); 
        
                $txtreleaseartwork =  $txtreleaseartwork_hdn;
                if($_FILES['txtreleaseartwork']['name'] != '')
                {
                     $ext = pathinfo($_FILES['txtreleaseartwork']['name'], PATHINFO_EXTENSION);
                     $file_name = date('YmdHis').'.'.$ext;
                     $config = array(
                        'upload_path' => 'uploaded',
                        'allowed_types' => 'gif|jpg|png|jpeg|pdf|docx|xls|xlsx|doc|txt',
                        'overwrite' => TRUE,
                        'file_name' =>  $file_name
                    );
                     $this->upload->initialize( $config);
                    if( $this->upload->do_upload('txtreleaseartwork'))
                    {
                         $data = $this->upload->data();
                        $txtreleaseartwork =  $file_name;
                        if($txtreleaseartwork_hdn != '')
                          unlink('uploaded/'.$txtreleaseartwork_hdn);
                    }
                    else
                    {
                        $data['msg'] = $this->upload->display_errors();
                        $data['type'] = 0;
                        echo json_encode($data);
                        exit();
                    }
                }
              

        if($txtstate == 'add')
        {
            $insert = array(
                'release_name' => $txtreleasename, 'release_date' => date("Y-m-d H:i:s", strtotime($txtreleasedate)), 'release_artwork' => $txtreleaseartwork, 'release_label' => $txtreleaselabel, 'release_upc' => $txtreleaseupc, 'release_taken' => date("Y-m-d H:i:s", strtotime($txtreleasetaken)), 'release_id' => $txtreleaseid,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'release_name' => $txtreleasename, 'release_date' => date("Y-m-d H:i:s", strtotime($txtreleasedate)), 'release_artwork' => $txtreleaseartwork, 'release_label' => $txtreleaselabel, 'release_upc' => $txtreleaseupc, 'release_taken' => date("Y-m-d H:i:s", strtotime($txtreleasetaken)), 'release_id' => $txtreleaseid,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }
