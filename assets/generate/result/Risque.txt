    public function save()
    {
        $txtstate = $this->input->post("txtstate");
        $txtrowid = $this->input->post("txtrowid");
        $txtRisque1 = $this->input->post("txtRisque1"); $txtRisque2 = $this->input->post("txtRisque2"); $txtRisque3 = $this->input->post("txtRisque3"); 
        

        if($txtstate == 'add')
        {
            $insert = array(
                'Risque1' => $txtRisque1, 'Risque2' => $txtRisque2, 'Risque3' => $txtRisque3,         
            );
            insertData($this->table_db, $insert);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
        else
        {            
            $update = array(
                'Risque1' => $txtRisque1, 'Risque2' => $txtRisque2, 'Risque3' => $txtRisque3,              
            );
            updateData($this->table_db, $update, "row_id = '".$txtrowid."'", $txtrowid);
            $data['msg'] = "Process successful";
            $data['type'] = 1;
            echo json_encode($data);
        }
    }

    public function delete(){
        $txtrowid = $this->input->post("txtrowid");

        $data['msg'] = "Process successful";
        $data['type'] = 1;
        echo json_encode($data);
